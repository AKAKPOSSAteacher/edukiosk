Number(row[5]) : null, score: row[6]!=='' ? txt(row[6]).split('/')[0] : null, max: row[6]!=='' ? txt(row[6]).split('/')[1] : null, duration: row[7]!=='' ? Number(row[7]) : null, abandoned: row[8]==='Oui', infractions: Number(row[9])||0 }); }); } var ps=getSheet('Parametres'); if(ps.getLastRow()===0){ps.appendRow(['cle','valeur']);ps.appendRow(['pin','PROF2024']);} if(ps.getLastRow()>1) ps.getRange(2,1,ps.getLastRow()-1,2).getValues().forEach(function(r){if(r[0]==='pin')pin=txt(r[1]);}); var result={ok:true,tests:tests,docs:docs,results:results,pin:pin}; var cb=e.parameter.callback; if(cb) return ContentService.createTextOutput(cb+'('+JSON.stringify(result)+')').setMimeType(ContentService.MimeType.JAVASCRIPT); return jsonResp(result); } catch(err){ return jsonResp({ok:false,error:err.toString()}); } } function writeResult(rs, r) { // S'assurer que la colonne C est en texte avant d'écrire var nextRow = rs.getLastRow() + 1; rs.getRange(nextRow, 3).setNumberFormat('@'); rs.getRange(nextRow, 1, 1, 10).setValues([[ txt(r.date) || new Date().toLocaleDateString('fr-FR'), txt(r.studentName), txt(r.classe), // texte pur txt(r.type) || 'test', txt(r.testTitle||r.docTitle), r.pct != null ? r.pct : '', r.score!= null ? txt(r.score)+'/'+txt(r.max) : '', r.duration != null ? r.duration : '', r.abandoned ? 'Oui' : 'Non', r.infractions || 0 ]]); } function doPost(e) { try { var body=JSON.parse(e.postData.contents); var action=body.action; if(action==='saveTests'){ var ts=getSheet('Tests'); ts.clearContents(); (body.data||[]).forEach(function(t){ts.appendRow([JSON.stringify(t)]);}); return jsonResp({ok:true,saved:(body.data||[]).length}); } if(action==='saveDocs'){ var ds=getSheet('Documents'); ds.clearContents(); (body.data||[]).forEach(function(d){ds.appendRow([JSON.stringify(d)]);}); return jsonResp({ok:true,saved:(body.data||[]).length}); } if(action==='saveResult'){ var rs=getSheet('Resultats'); setupResultsSheet(rs); writeResult(rs, body.data); return jsonResp({ok:true}); } if(action==='saveResults'){ var rs2=getSheet('Resultats'); rs2.clearContents(); rs2.appendRow(['Date','Eleve','Classe','Type','Titre','Score %','Score brut','Duree (s)','Abandonne','Infractions']); rs2.getRange(1,1,1,10).setFontWeight('bold').setBackground('#0f2044').setFontColor('#ffffff'); rs2.getRange('C:C').setNumberFormat('@'); (body.data||[]).forEach(function(r){writeResult(rs2,r);}); return jsonResp({ok:true,saved:(body.data||[]).length}); } if(action==='savePin'){ var ps=getSheet('Parametres'); var found=false; if(ps.getLastRow()>1) ps.getRange(2,1,ps.getLastRow()-1,2).getValues().forEach(function(r,i){if(r[0]==='pin'){ps.getRange(i+2,2).setValue(body.data);found=true;}}); if(!found) ps.appendRow(['pin',body.data]); return jsonResp({ok:true}); } return jsonResp({ok:false,error:'Action inconnue: '+action}); } catch(err){ return jsonResp({ok:false,error:err.toString()}); } }
